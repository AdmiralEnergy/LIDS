# ADS Dashboard Progression System - Replit Implementation Prompt

## What You're Building

A AAA video game-like progression system for sales reps that makes invisible growth visible. This feature integrates into the existing ADS Dashboard (LIDS codebase).

**Core Philosophy:** "A great game makes you feel skilled, not manipulated."

## Architecture: LOCAL-FIRST (No Backend Dependencies)

**CRITICAL:** This system runs entirely in the browser with NO backend requirements.

- Works in Replit's contained environment
- No API keys or auth tokens needed
- Instant - no network latency
- Works offline
- Data persists in browser

The ADS Dashboard already has:

## Feature Overview

### 1. XP & Leveling System
- Every activity earns XP (dials, connects, appointments, deals)
- Level 1-15+ progression with exponential thresholds
- Floating "+XP" animations when XP earned
- Level-up celebrations

### 2. Achievement Badges
- 4 tiers: Bronze, Silver, Gold, Platinum
- Categories: Communication, Performance, Streaks, Special
- Visual badge display with progress tracking
- Unlock animations with confetti

### 3. Rank System (SDR to Sales Operative)
- SDR I (E-1) â†’ SDR II (E-2) â†’ SDR III (E-3) â†’ Sales Operative (E-4)
- Skill gates: XP requirements, badge requirements, exams

### 4. Specialization Paths (Unlocked at Level 5)
- Speed Dialer: +25% dial XP
- Relationship Builder: +50% appointment XP
- Closer Mindset: +75% deal XP
- Referral Master: +100% referral XP

### 5. Leaderboards
- Daily, Weekly, All-Time boards
- Category champions (Cold Calling, Email, Closer, Consistency)
- Real-time updates

## Local Storage Schema (Dexie.js)

**Install Dexie:** `npm install dexie`

Create `src/lib/progressionDb.ts`:

```typescript
import Dexie, { Table } from 'dexie';

// Types
export interface UserProgression {
  id: string;  // 'current' for single-user, or odrer ID for multi-user
  name: string;
  rank: string;  // sdr_1, sdr_2, sdr_3, operative, senior_operative
  totalXp: number;
  currentLevel: number;
  closedDeals: number;
  graduationDate?: Date;
  badges: string[];  // Array of badge IDs like 'cold_calling_master.bronze'
  specialization?: string;  // speed_dialer, relationship, closer, referral
  streakDays: number;
  lastActivityDate: Date;
}

export interface XPEvent {
  id?: number;
  eventType: string;
  xpAmount: number;
  multipliers: Record<string, number>;
  createdAt: Date;
}

export interface BadgeProgress {
  badgeId: string;
  currentTier?: string;
  progress: Record<string, number>;  // { dials: 500, connect_rate: 0.15 }
  earnedAt?: Date;
}

export interface ActivityLog {
  id?: number;
  action: string;
  details: string;
  xpEarned: number;
  timestamp: Date;
}

// Database
class ProgressionDatabase extends Dexie {
  progression!: Table<UserProgression>;
  xpEvents!: Table<XPEvent>;
  badgeProgress!: Table<BadgeProgress>;
  activityLog!: Table<ActivityLog>;

  constructor() {
    super('ADS_Progression');
    this.version(1).stores({
      progression: 'id',
      xpEvents: '++id, eventType, createdAt',
      badgeProgress: 'badgeId',
      activityLog: '++id, action, timestamp',
    });
  }
}

export const db = new ProgressionDatabase();

// Initialize default progression if not exists
export async function initProgression(name: string = 'Sales Rep') {
  const existing = await db.progression.get('current');
  if (!existing) {
    await db.progression.put({
      id: 'current',
      name,
      rank: 'sdr_1',
      totalXp: 0,
      currentLevel: 1,
      closedDeals: 0,
      badges: [],
      streakDays: 0,
      lastActivityDate: new Date(),
    });
  }
  return db.progression.get('current');
}

// Export for backup
export async function exportProgressionData() {
  return {
    progression: await db.progression.toArray(),
    xpEvents: await db.xpEvents.toArray(),
    badgeProgress: await db.badgeProgress.toArray(),
    activityLog: await db.activityLog.toArray(),
    exportedAt: new Date().toISOString(),
  };
}

// Import from backup
export async function importProgressionData(data: any) {
  await db.progression.clear();
  await db.xpEvents.clear();
  await db.badgeProgress.clear();
  await db.activityLog.clear();

  if (data.progression) await db.progression.bulkPut(data.progression);
  if (data.xpEvents) await db.xpEvents.bulkPut(data.xpEvents);
  if (data.badgeProgress) await db.badgeProgress.bulkPut(data.badgeProgress);
  if (data.activityLog) await db.activityLog.bulkPut(data.activityLog);
}
```

### 3. useProgression.ts (Main Hook - Local Storage)

```tsx
import { useState, useEffect, useCallback } from 'react';
import { useLiveQuery } from 'dexie-react-hooks';
import { db, initProgression, UserProgression } from '@/lib/progressionDb';
import { XP_THRESHOLDS, RANKS } from '../config';

interface AddXPParams {
  eventType: string;
  amount: number;
  multipliers?: Record<string, number>;
  details?: string;
}

export function useProgression() {
  const [isLoading, setIsLoading] = useState(true);

  // Live query from IndexedDB - auto-updates when data changes
  const progression = useLiveQuery(() => db.progression.get('current'));
  const recentEvents = useLiveQuery(() =>
    db.xpEvents.orderBy('createdAt').reverse().limit(10).toArray()
  );

  // Initialize on mount
  useEffect(() => {
    initProgression().then(() => setIsLoading(false));
  }, []);

  // Calculate level from total XP
  const level = calculateLevel(progression?.totalXp || 0);
  const xpForCurrentLevel = XP_THRESHOLDS[level - 1] || 0;
  const xpForNextLevel = XP_THRESHOLDS[level] || XP_THRESHOLDS[XP_THRESHOLDS.length - 1];
  const xpProgress = (progression?.totalXp || 0) - xpForCurrentLevel;
  const xpToNextLevel = xpForNextLevel - xpForCurrentLevel;

  // Add XP function
  const addXP = useCallback(async ({ eventType, amount, multipliers = {}, details = '' }: AddXPParams) => {
    const finalAmount = applyMultipliers(amount, multipliers);
    const current = await db.progression.get('current');
    if (!current) return 0;

    const newXp = current.totalXp + finalAmount;
    const newLevel = calculateLevel(newXp);

    // Update progression
    await db.progression.update('current', {
      totalXp: newXp,
      currentLevel: newLevel,
      lastActivityDate: new Date(),
    });

    // Log XP event
    await db.xpEvents.add({
      eventType,
      xpAmount: finalAmount,
      multipliers,
      createdAt: new Date(),
    });

    // Log activity
    await db.activityLog.add({
      action: eventType,
      details,
      xpEarned: finalAmount,
      timestamp: new Date(),
    });

    // Check for level up
    if (newLevel > current.currentLevel) {
      // Trigger level up event (handle in component)
      window.dispatchEvent(new CustomEvent('levelUp', {
        detail: { oldLevel: current.currentLevel, newLevel }
      }));
    }

    return finalAmount;
  }, []);

  // Update rank
  const updateRank = useCallback(async (newRank: string) => {
    await db.progression.update('current', {
      rank: newRank,
      graduationDate: newRank === 'operative' ? new Date() : undefined,
    });
  }, []);

  // Add badge
  const addBadge = useCallback(async (badgeId: string) => {
    const current = await db.progression.get('current');
    if (!current) return;

    const badges = [...current.badges, badgeId];
    await db.progression.update('current', { badges });

    // Log badge progress
    await db.badgeProgress.put({
      badgeId,
      currentTier: badgeId.split('.')[1],
      progress: {},
      earnedAt: new Date(),
    });

    // Trigger badge unlock event
    window.dispatchEvent(new CustomEvent('badgeUnlock', { detail: { badgeId } }));
  }, []);

  // Set specialization
  const setSpecialization = useCallback(async (path: string) => {
    await db.progression.update('current', { specialization: path });
  }, []);

  return {
    progression,
    level,
    xp: progression?.totalXp || 0,
    xpProgress,
    xpToNextLevel,
    xpPercent: Math.round((xpProgress / xpToNextLevel) * 100),
    rank: RANKS[progression?.rank || 'sdr_1'],
    badges: progression?.badges || [],
    specialization: progression?.specialization,
    recentEvents: recentEvents || [],
    isLoading,
    addXP,
    updateRank,
    addBadge,
    setSpecialization,
  };
}

function calculateLevel(xp: number): number {
  for (let i = XP_THRESHOLDS.length - 1; i >= 0; i--) {
    if (xp >= XP_THRESHOLDS[i]) return i + 1;
  }
  return 1;
}

function applyMultipliers(base: number, multipliers: Record<string, number>): number {
  let total = base;
  for (const mult of Object.values(multipliers)) {
    total *= mult;
  }
  return Math.round(total);
}
```

## Success Criteria

1. **XP System:** Activities generate XP, levels increase correctly
2. **Badges:** Reps can see badges with progress toward next tier
3. **Promotion Gates:** Clear requirements visible for next rank
4. **Leaderboards:** Daily/weekly rankings update in real-time
5. **Animations:** XP floaters, level-up celebrations, badge unlocks
6. **Mobile:** Works on tablets and phones

---

## Don't Do

- Don't add Supabase auth (already exists)
- Don't add complex state management (use React Query)
- Don't over-engineer the XP calculation
- Don't add real money or paid features
- Keep animations subtle, not distracting during calls
